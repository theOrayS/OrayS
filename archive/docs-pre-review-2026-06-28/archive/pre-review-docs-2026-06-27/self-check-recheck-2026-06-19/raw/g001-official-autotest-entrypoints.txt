   124	def parse_iperf(output):
   125	    pat = re.compile(r"^\[\s*[56SUM]*].+[\d.]+ [MG]bits/sec.+receiver$")
   126	    key = ""
   127	    ans = {}
   128	    for line in output.split("\n"):
   129	        if "====== iperf BASIC_UDP begin ======" in line:
   130	            key = "iperf BASIC_UDP"
   131	        elif "====== iperf BASIC_TCP begin ======" in line:
   132	            key = "iperf BASIC_TCP"
   133	        elif "====== iperf PARALLEL_UDP begin ======" in line:
   134	            key = "iperf PARALLEL_UDP"
   135	        elif "====== iperf PARALLEL_TCP begin ======" in line:
   136	            key = "iperf PARALLEL_TCP"
   137	        elif "====== iperf REVERSE_UDP begin ======" in line:
   138	            key = "iperf REVERSE_UDP"
   139	        elif "====== iperf REVERSE_TCP begin ======" in line:
   140	            key = "iperf REVERSE_TCP"
   141	        res = pat.findall(line)
   142	        if res:
   143	            res = res[0].split()
   144	            for i in range(0, len(res)):
   145	                if res[i] == "Mbits/sec" or res[i] == "Gbits/sec":
   146	                    val = float(res[i-1])
   147	                    if res[i] == "Gbits/sec":
   148	                        val = val * 1024
   149	                    ans[key + " " + "(Mbits/sec)"] = val
   150	                    break
   151	    return ans
   152
   153	def generate_score(results, baseline):
   154	    lmbench_results = results
   155	    lmbench_baseline = baseline
   156	    lmbench = [
   157	        {
   158	            "name": name,
   159	            "res": lmbench_results.get(name, 0),
   160	            "baseline": baseline,
   161	            "score": 0.0
   162	        }
   163	        for name, baseline in lmbench_baseline.items()
   164	    ]
   165	    for item in lmbench:
   166	        if item["res"] > 0:
   167	            if "microseconds" in item["name"] or "seconds" in item["name"]:
   168	                item["score"] = item["baseline"] / item["res"]
   169	            else:
   170	                item["score"] = item["res"] / item["baseline"]
   171
   172	            if item['score'] >= 1:
   173	                item['score'] = 2 - (1 / item['score'])
   174	            else:
   175	                item['score'] = 1.0
   176	    return lmbench
   177

    41	def parse_ltp_log(content: str):
    42	    lines = content.split('\n')
    43	    result = {}
    44	    testcase = None
    45	    passed, failed, broken, skipped, warnings = 0, 0, 0, 0, 0
    46	    for line in lines:
    47	        line = line.strip()
    48
    49	        if line.startswith('RUN LTP CASE'):
    50	            testcase = line.split()[-1]
    51	            continue
    52
    53	        # Look at LTP logs (TPASS, TFAIL etc.) rather than summary.
    54	        # Some LTP binaries don't produce summaries when run directly.
    55	        if line.startswith("FAIL LTP CASE"):
    56	            result[testcase] = {
    57	                "success": passed,
    58	                "failed": failed,
    59	                "broken": broken,
    60	                "skipped": skipped,
    61	                "warnings": warnings,
    62	                "all": passed + failed + broken + skipped + warnings
    63	            }
    64	            testcase = ""
    65	            passed, failed, broken, skipped, warnings = 0, 0, 0, 0, 0
    66	            continue
    67
    68	        if "\x1b[1;32mTPASS: \x1b[0m" in line:
    69	            passed += 1
    70	            continue
    71	        if "\x1b[1;31mTFAIL: \x1b[0m" in line:
    72	            failed += 1
    73	            continue
    74	        if "\x1b[1;31mTBROK: \x1b[0m" in line:
    75	            broken += 1
    76	            continue
    77	        if "\x1b[1;33mTCONF: \x1b[0m" in line:
    78	            skipped += 1
    79	            continue
    80	        if "\x1b[1;35mTWARN: \x1b[0m" in line:

    21	The auxiliary test data is used for judging your work. You can choose any directory you like, which we call `$data` in the following script.
    22
    23	```bash
    24	# Create the directory.
    25	mkdir $data
    26
    27	# Copy the judge scripts there.
    28	cd autotest-for-oskernel
    29	cp -rf kernel/judge/* $data
    30
    31	# Download the SD card images.
    32	cd $data
    33	wget https://github.com/oscomp/testsuits-for-oskernel/releases/download/pre-20250615/sdcard-la.img.xz
    34	wget https://github.com/oscomp/testsuits-for-oskernel/releases/download/pre-20250615/sdcard-rv.img.xz
    35
    36	# It is possible to omit the `gzip` in order to save time,
    37	# provided you modify the testing scripts locally.
    38	unxz sdcard-la.img.xz
    39	gzip sdcard-la.img
    40	unxz sdcard-rv.img.xz
    41	gzip sdcard-rv.img
    42	```
    43
    44	## 4. Prepare auxiliary python kernel for judge
    45
    46	Note this "kernel" is not the operating system kernel.
    47
    48	```bash
    49	cd autotest-for-oskernel/kernel
    50	zip ../kernel.zip -r *
    51	```
    52
    53	## 5. Evaluate your work
    54
    55	We assume you already have your work locally. We refer to its folder as `$os` in the following script.
    56
    57	Navigate to the parent folder of this repository, and run:
    58
    59	```bash
    60	sudo docker run --rm \
    61	-v $os:/coursegrader/submit \
    62	-v $data:/coursegrader/testdata \
    63	-v autotest-for-oskernel:/cg \
    64	-v $data:/mnt/cghook/ \
    65	zhouzhouyi/os-contest:20260510 python3 /cg/kernel.zip
    66	```
    67
    68	The docker will build your OS, evaluate it, and output the result on the console.
    69
    70	To stop it, use `docker stop`, rather than force-exiting the python script. The latter method will leave the files in a locked state, and unlocking might require a reboot.
