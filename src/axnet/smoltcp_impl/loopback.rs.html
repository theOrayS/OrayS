9const LOOPBACK_BUFFER_LIMIT: usize = 1024 * 1024;
10
11fn loopback_buffer_limit(sender_send_size: usize, receiver_recv_size: usize) -> usize {
12    min(
13        min(sender_send_size, receiver_recv_size),
14        LOOPBACK_BUFFER_LIMIT,
15    )
16}

18struct LoopbackTcpState {
19    client_to_server: VecDeque<u8>,
20    server_to_client: VecDeque<u8>,
21    client_to_server_limit: usize,
22    server_to_client_limit: usize,
23    client_send_open: bool,
24    server_send_open: bool,
25    client_recv_open: bool,
26    server_recv_open: bool,
27}

29#[derive(Clone)]
30pub(crate) struct LoopbackTcpEndpoint {
31    state: Arc<Mutex<LoopbackTcpState>>,
32    server_side: bool,
33    local_addr: IpEndpoint,
34    peer_addr: IpEndpoint,
35}

37impl LoopbackTcpEndpoint {
38    pub(crate) fn pair(
39        client_addr: IpEndpoint,
40        server_addr: IpEndpoint,
41        client_recv_buffer_size: usize,
42        client_send_buffer_size: usize,
43        server_recv_buffer_size: usize,
44        server_send_buffer_size: usize,
45    ) -> (LoopbackTcpEndpoint, LoopbackTcpEndpoint) {
46        let state = Arc::new(Mutex::new(LoopbackTcpState {
47            client_to_server: VecDeque::new(),
48            server_to_client: VecDeque::new(),
49            client_to_server_limit: loopback_buffer_limit(
50                client_send_buffer_size,
51                server_recv_buffer_size,
52            ),
53            server_to_client_limit: loopback_buffer_limit(
54                server_send_buffer_size,
55                client_recv_buffer_size,
56            ),
57            client_send_open: true,
58            server_send_open: true,
59            client_recv_open: true,
60            server_recv_open: true,
61        }));
62        (
63            LoopbackTcpEndpoint {
64                state: state.clone(),
65                server_side: false,
66                local_addr: client_addr,
67                peer_addr: server_addr,
68            },
69            LoopbackTcpEndpoint {
70                state,
71                server_side: true,
72                local_addr: server_addr,
73                peer_addr: client_addr,
74            },
75        )
76    }
77
78    pub(crate) fn local_addr(&self) -> IpEndpoint {
79        self.local_addr
80    }
81
82    pub(crate) fn peer_addr(&self) -> IpEndpoint {
83        self.peer_addr
84    }

86    pub(crate) fn send(&self, buf: &[u8]) -> AxResult<usize> {
87        if buf.is_empty() {
88            return Ok(0);
89        }
90
91        let mut state = self.state.lock();
92        let local_send_open = if self.server_side {
93            state.server_send_open
94        } else {
95            state.client_send_open
96        };
97        let peer_recv_open = if self.server_side {
98            state.client_recv_open
99        } else {
100            state.server_recv_open
101        };
102        if !local_send_open || !peer_recv_open {
103            return ax_err!(BrokenPipe, "loopback TCP send() failed");
104        }
105
106        let (tx_queue, tx_limit) = if self.server_side {
107            let tx_limit = state.server_to_client_limit;
108            (&mut state.server_to_client, tx_limit)
109        } else {
110            let tx_limit = state.client_to_server_limit;
111            (&mut state.client_to_server, tx_limit)
112        };
113        if tx_queue.len() >= tx_limit {
114            return Err(AxError::WouldBlock);
115        }
116
117        let len = min(buf.len(), tx_limit - tx_queue.len());
118        tx_queue.extend(buf[..len].iter().copied());
119        Ok(len)
120    }
121
122    pub(crate) fn recv(&self, buf: &mut [u8]) -> AxResult<usize> {
123        if buf.is_empty() {
124            return Ok(0);
125        }
126
127        let mut state = self.state.lock();
128        let local_recv_open = if self.server_side {
129            state.server_recv_open
130        } else {
131            state.client_recv_open
132        };
133        if !local_recv_open {
134            return Ok(0);
135        }
136        let peer_send_open = if self.server_side {
137            state.client_send_open
138        } else {
139            state.server_send_open
140        };
141        let rx_queue = if self.server_side {
142            &mut state.client_to_server
143        } else {
144            &mut state.server_to_client
145        };
146
147        if rx_queue.is_empty() {
148            return if peer_send_open {
149                Err(AxError::WouldBlock)
150            } else {
151                Ok(0)
152            };
153        }
154
155        let len = min(buf.len(), rx_queue.len());
156        for slot in buf.iter_mut().take(len) {
157            *slot = rx_queue.pop_front().unwrap();
158        }
159        Ok(len)
160    }
161
162    pub(crate) fn poll(&self) -> PollState {
163        let state = self.state.lock();
164        let (local_recv_open, peer_send_open, peer_recv_open, rx_len, tx_len, tx_limit) =
165            if self.server_side {
166                (
167                    state.server_recv_open,
168                    state.client_send_open,
169                    state.client_recv_open,
170                    state.client_to_server.len(),
171                    state.server_to_client.len(),
172                    state.server_to_client_limit,
173                )
174            } else {
175                (
176                    state.client_recv_open,
177                    state.server_send_open,
178                    state.server_recv_open,
179                    state.server_to_client.len(),
180                    state.client_to_server.len(),
181                    state.client_to_server_limit,
182                )
183            };
184        PollState {
185            readable: rx_len > 0 || !local_recv_open || !peer_send_open,
186            writable: peer_recv_open && tx_len < tx_limit,
187        }
188    }
189
190    pub(crate) fn shutdown_read(&self) {
191        let mut state = self.state.lock();
192        if self.server_side {
193            state.server_recv_open = false;
194        } else {
195            state.client_recv_open = false;
196        }
197    }
198
199    pub(crate) fn shutdown_write(&self) {
200        let mut state = self.state.lock();
201        if self.server_side {
202            state.server_send_open = false;
203        } else {
204            state.client_send_open = false;
205        }
206    }
207
208    pub(crate) fn shutdown(&self) {
209        let mut state = self.state.lock();
210        if self.server_side {
211            state.server_send_open = false;
212            state.server_recv_open = false;
213        } else {
214            state.client_send_open = false;
215            state.client_recv_open = false;
216        }
217    }
