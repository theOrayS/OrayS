14fn remove_waiters_matching<F>(queue: &mut VecDeque<AxTaskRef>, mut should_remove: F) -> bool
15where
16    F: FnMut(&AxTaskRef) -> bool,
17{
18    let len = queue.len();
19    let mut removed = false;
20    for _ in 0..len {
21        let Some(task) = queue.pop_front() else {
22            break;
23        };
24        if should_remove(&task) {
25            removed = true;
26        } else {
27            queue.push_back(task);
28        }
29    }
30    removed
31}

33/// A queue to store sleeping tasks.
34///
35/// # Examples
36///
37/// ```
38/// use axtask::WaitQueue;
39/// use core::sync::atomic::{AtomicU32, Ordering};
40///
41/// static VALUE: AtomicU32 = AtomicU32::new(0);
42/// static WQ: WaitQueue = WaitQueue::new();
43///
44/// axtask::init_scheduler();
45/// // spawn a new task that updates `VALUE` and notifies the main task
46/// axtask::spawn(|| {
47///     assert_eq!(VALUE.load(Ordering::Acquire), 0);
48///     VALUE.fetch_add(1, Ordering::Release);
49///     WQ.notify_one(true); // wake up the main task
50/// });
51///
52/// WQ.wait(); // block until `notify()` is called
53/// assert_eq!(VALUE.load(Ordering::Acquire), 1);
54/// ```
55pub struct WaitQueue {
56    queue: SpinNoIrq<VecDeque<AxTaskRef>>,
57}

59pub(crate) type WaitQueueGuard<'a> = SpinNoIrqGuard<'a, VecDeque<AxTaskRef>>;

76    /// Cancel events by removing the task from the wait queue.
77    /// If `from_timer_list` is true, try to remove the task from the timer list.
78    fn cancel_events(&self, curr: CurrentTask, _from_timer_list: bool) {
79        // A task can be wake up only one events (timer or `notify()`), remove
80        // the event from another queue.
81        if curr.in_wait_queue() {
82            // wake up by timer (timeout).
83            let mut wq = self.queue.lock();
84            remove_waiters_matching(&mut wq, |t| curr.ptr_eq(t));
85            curr.set_in_wait_queue(false);
86        }
87
88        // Try to cancel a timer event from timer lists and mark the current
89        // ticket as expired. Timed waits that are woken by notify/signals may
90        // otherwise leave future-deadline events in the timer heap; workloads
91        // with many short timed waits (for example pthread condition-variable
92        // timeouts) can grow that heap until later allocations fail.
93        #[cfg(feature = "irq")]
94        if _from_timer_list {
95            crate::timers::cancel_alarm_wakeup(curr.as_task_ref());
96            curr.timer_ticket_expired();
97        }
98    }

100    /// Blocks the current task and put it into the wait queue, until other task
101    /// notifies it.
102    pub fn wait(&self) {
103        current_run_queue::<NoPreemptIrqSave>().blocked_resched(self.queue.lock());
104        self.cancel_events(crate::current(), false);
105    }

107    /// Blocks the current task and put it into the wait queue, until the given
108    /// `condition` becomes true.
109    ///
110    /// Note that even other tasks notify this task, it will not wake up until
111    /// the condition becomes true.
112    pub fn wait_until<F>(&self, condition: F)
113    where
114        F: Fn() -> bool,
115    {
116        let curr = crate::current();
117        loop {
118            let mut rq = current_run_queue::<NoPreemptIrqSave>();
119            let wq = self.queue.lock();
120            if condition() {
121                break;
122            }
123            rq.blocked_resched(wq);
124            // Preemption may occur here.
125        }
126        self.cancel_events(curr, false);
127    }

129    /// Blocks the current task and put it into the wait queue, until other tasks
130    /// notify it, or the given duration has elapsed.
131    #[cfg(feature = "irq")]
132    pub fn wait_timeout(&self, dur: core::time::Duration) -> bool {
133        if dur.is_zero() {
134            return true;
135        }
136        let mut rq = current_run_queue::<NoPreemptIrqSave>();
137        let curr = crate::current();
138        let deadline = axhal::time::wall_time() + dur;
139        debug!(
140            "task wait_timeout: {} deadline={:?}",
141            curr.id_name(),
142            deadline
143        );
144        crate::timers::set_alarm_wakeup(deadline, curr.clone());
145
146        rq.blocked_resched(self.queue.lock());
147
148        let timeout = curr.in_wait_queue(); // still in the wait queue, must have timed out
149
150        // Always try to remove the task from the timer list.
151        self.cancel_events(curr, true);
152        timeout
153    }

155    /// Blocks the current task and put it into the wait queue, until the given
156    /// `condition` becomes true, or the given duration has elapsed.
157    ///
158    /// Note that even other tasks notify this task, it will not wake up until
159    /// the above conditions are met.
160    #[cfg(feature = "irq")]
161    pub fn wait_timeout_until<F>(&self, dur: core::time::Duration, condition: F) -> bool
162    where
163        F: Fn() -> bool,
164    {
165        let curr = crate::current();
166        if condition() {
167            return false;
168        }
169        if dur.is_zero() {
170            return true;
171        }
172        let deadline = axhal::time::wall_time() + dur;
173        if axhal::time::wall_time() >= deadline {
174            return true;
175        }
176        debug!(
177            "task wait_timeout: {}, deadline={:?}",
178            curr.id_name(),
179            deadline
180        );
181        crate::timers::set_alarm_wakeup(deadline, curr.clone());
182
183        let mut timeout = true;
184        loop {
185            let mut rq = current_run_queue::<NoPreemptIrqSave>();
186            if axhal::time::wall_time() >= deadline {
187                break;
188            }
189            let wq = self.queue.lock();
190            if condition() {
191                timeout = false;
192                break;
193            }
194
195            rq.blocked_resched(wq);
196            // Preemption may occur here.
197        }
198        // Always try to remove the task from the timer list.
199        self.cancel_events(curr, true);
200        timeout
201    }

203    /// Wakes up one task in the wait queue, usually the first one.
204    ///
205    /// If `resched` is true, the current task will be preempted when the
206    /// preemption is enabled.
207    pub fn notify_one(&self, resched: bool) -> bool {
208        let mut wq = self.queue.lock();
209        // Timed waits can be woken by their timer before a notifier reaches the
210        // queued entry. Drain such stale entries until one task actually moves
211        // from Blocked to Ready.
212        while let Some(task) = wq.pop_front() {
213            if unblock_one_task(task, resched) {
214                return true;
215            }
216        }
217        false
218    }

220    /// Wakes up to `count` distinct tasks, ignoring duplicate queue entries for
221    /// tasks already selected by this call.
222    ///
223    /// Timed waits may leave stale duplicate entries if a task is awakened and
224    /// re-checks a false condition before its final timeout/cancel cleanup.  For
225    /// futex wake, the syscall return value is part of the userspace protocol, so
226    /// it must count distinct waiters rather than raw queue nodes.
227    pub fn notify_many_unique(&self, count: usize, resched: bool) -> usize {
228        if count == 0 {
229            return 0;
230        }
231
232        let mut notified = BTreeSet::new();
233        let mut wq = self.queue.lock();
234        let mut notified_len = 0usize;
235        while notified_len < count {
236            let Some(task) = wq.pop_front() else {
237                break;
238            };
239            if !notified.insert(task_ptr_key(&task)) {
240                continue;
241            }
242            if unblock_one_task(task, resched) {
243                notified_len += 1;
244            }
245        }
246        notified_len
247    }

249    /// Wakes up to `count` distinct tasks that satisfy `predicate`, invoking
250    /// `on_task` only for tasks that are actually moved from blocked to ready.
251    pub fn notify_many_where<F, G>(
252        &self,
253        count: usize,
254        resched: bool,
255        mut predicate: F,
256        mut on_task: G,
257    ) -> usize
258    where
259        F: FnMut(&AxTaskRef) -> bool,
260        G: FnMut(&AxTaskRef),
261    {
262        if count == 0 {
263            return 0;
264        }
265
266        let mut selected = BTreeSet::new();
267        let mut notified_len = 0usize;
268        let mut wq = self.queue.lock();
269        let mut index = 0;
270        while index < wq.len() && notified_len < count {
271            let Some(task) = wq.get(index).cloned() else {
272                break;
273            };
274            let key = task_ptr_key(&task);
275            if selected.contains(&key) {
276                let _ = wq.remove(index);
277                continue;
278            }
279            if !predicate(&task) {
280                index += 1;
281                continue;
282            }
283            let Some(task) = wq.remove(index) else {
284                break;
285            };
286            selected.insert(key);
287            if unblock_one_task_with(task, resched, |task| on_task(task)) {
288                notified_len += 1;
289            }
290        }
291        notified_len
292    }

294    /// Wakes up to `wake_count` distinct source waiters matching `predicate`
295    /// and requeues up to `requeue_count` remaining distinct source waiters to
296    /// `target` while holding every involved queue lock.
297    ///
298    /// Futex requeue users rely on the wake and requeue phases being a single
299    /// handoff: a concurrent wake on the target queue must not run between
300    /// removing waiters from the source and inserting them into the target.
301    pub fn notify_and_requeue_where<F, G, H>(
302        &self,
303        wake_count: usize,
304        requeue_count: usize,
305        target: &WaitQueue,
306        resched: bool,
307        mut predicate: F,
308        mut on_wake: G,
309        mut on_requeue: H,
310    ) -> (usize, usize)
311    where
312        F: FnMut(&AxTaskRef) -> bool,
313        G: FnMut(&AxTaskRef),
314        H: FnMut(&AxTaskRef),
315    {
316        if wake_count == 0 && requeue_count == 0 {
317            return (0, 0);
318        }
319
320        let mut operate =
321            |source: &mut VecDeque<AxTaskRef>,
322             mut destination: Option<&mut VecDeque<AxTaskRef>>| {
323                let mut selected = BTreeSet::new();
324                let mut notified_len = 0usize;
325                let mut index = 0;
326                while index < source.len() && notified_len < wake_count {
327                    let Some(task) = source.get(index).cloned() else {
328                        break;
329                    };
330                    let key = task_ptr_key(&task);
331                    if selected.contains(&key) {
332                        let _ = source.remove(index);
333                        continue;
334                    }
335                    if !predicate(&task) {
336                        index += 1;
337                        continue;
338                    }
339                    let Some(task) = source.remove(index) else {
340                        break;
341                    };
342                    selected.insert(key);
343                    if unblock_one_task_with(task, resched, |task| on_wake(task)) {
344                        notified_len += 1;
345                    }
346                }
347
348                let mut requeued = BTreeSet::new();
349                let mut requeued_len = 0usize;
350                if let Some(destination) = destination.as_deref_mut() {
351                    while !source.is_empty() && requeued_len < requeue_count {
352                        let Some(task) = source.pop_front() else {
353                            break;
354                        };
355                        let key = task_ptr_key(&task);
356                        if selected.contains(&key) || !requeued.insert(key) {
357                            continue;
358                        }
359                        on_requeue(&task);
360                        destination.push_back(task);
361                        requeued_len += 1;
362                    }
363                }
364                (notified_len, requeued_len)
365            };
366
367        if core::ptr::eq(self, target) {
368            let mut source = self.queue.lock();
369            return operate(&mut source, None);
370        }
371
372        let self_addr = core::ptr::addr_of!(self.queue) as usize;
373        let target_addr = core::ptr::addr_of!(target.queue) as usize;
374        if self_addr < target_addr {
375            let mut source = self.queue.lock();
376            let mut destination = target.queue.lock();
377            operate(&mut source, Some(&mut destination))
378        } else {
379            let mut destination = target.queue.lock();
380            let mut source = self.queue.lock();
381            operate(&mut source, Some(&mut destination))
382        }
383    }

395    /// Wake up the given task in the wait queue.
396    ///
397    /// If `resched` is true, the current task will be preempted when the
398    /// preemption is enabled.
399    pub fn notify_task(&self, resched: bool, task: &AxTaskRef) -> bool {
400        let mut wq = self.queue.lock();
401        if let Some(index) = wq.iter().position(|t| Arc::ptr_eq(t, task)) {
402            unblock_one_task(wq.remove(index).unwrap(), resched)
403        } else {
404            false
405        }
406    }

408    /// Removes `task` from this wait queue without waking it.
409    ///
410    /// Higher-level wait protocols use this to clean up after a task was moved
411    /// to a different queue and then timed out or was interrupted before that
412    /// destination queue woke it.
413    pub fn remove_task(&self, task: &AxTaskRef) -> bool {
414        let mut wq = self.queue.lock();
415        remove_waiters_matching(&mut wq, |queued| Arc::ptr_eq(queued, task))
416    }

418    /// Transfers up to `count` tasks from this wait queue to another wait queue.
419    ///
420    /// Note: If the current wait queue contains fewer than `count` tasks, all available tasks will be moved.
421    ///
422    /// ## Arguments
423    /// * `count` - The maximum number of tasks to be moved.
424    /// * `target` - The target wait queue to which tasks will be moved.
425    ///
426    /// ## Returns
427    /// The number of tasks actually requeued.  
428    pub fn requeue(&self, count: usize, target: &WaitQueue) -> usize {
429        self.requeue_with(count, target, |_| {})
430    }

432    /// Transfers up to `count` tasks from this wait queue to another wait
433    /// queue, invoking `on_task` while each task is still owned by the requeue
434    /// operation.  This lets higher-level wait protocols update per-task wait
435    /// metadata atomically with the queue transfer instead of pretending a
436    /// requeue was just a wake-up.
437    pub fn requeue_with<F>(&self, count: usize, target: &WaitQueue, mut on_task: F) -> usize
438    where
439        F: FnMut(&AxTaskRef),
440    {
441        if core::ptr::eq(self, target) || count == 0 {
442            return 0;
443        }
444
445        let mut transfer = |source: &mut VecDeque<AxTaskRef>,
446                            destination: &mut VecDeque<AxTaskRef>| {
447            let limit = count.min(source.len());
448            let mut moved = 0;
449            for _ in 0..limit {
450                let Some(task) = source.pop_front() else {
451                    break;
452                };
453                on_task(&task);
454                destination.push_back(task);
455                moved += 1;
456            }
457            moved
458        };
459
460        // Hold both queues while moving tasks so a concurrent wake on the
461        // target futex cannot miss waiters between source removal and
462        // destination insertion.  Lock addresses impose a stable order for
463        // simultaneous cross-requeues.
464        let self_addr = core::ptr::addr_of!(self.queue) as usize;
465        let target_addr = core::ptr::addr_of!(target.queue) as usize;
466        if self_addr < target_addr {
467            let mut source = self.queue.lock();
468            let mut destination = target.queue.lock();
469            transfer(&mut source, &mut destination)
470        } else {
471            let mut destination = target.queue.lock();
472            let mut source = self.queue.lock();
473            transfer(&mut source, &mut destination)
474        }
475    }

488fn unblock_one_task(task: AxTaskRef, resched: bool) -> bool {
489    unblock_one_task_with(task, resched, |_| {})
490}

492fn unblock_one_task_with<F>(task: AxTaskRef, resched: bool, on_ready: F) -> bool
493where
494    F: FnOnce(&AxTaskRef),
495{
496    // Mark task as not in wait queue.
497    task.set_in_wait_queue(false);
498    // Select run queue by the CPU set of the task.
499    // Use `NoOp` kernel guard here because the function is called with holding the
500    // lock of wait queue, where the irq and preemption are disabled.
501    select_run_queue::<NoOp>(&task).unblock_task_with(task, resched, on_ready)
502}
