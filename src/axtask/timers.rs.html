14static TIMER_TICKET_ID: AtomicU64 = AtomicU64::new(1);
15const MIN_ONESHOT_DELAY_NANOS: u64 = 1_000_000;
16
17percpu_static! {
18    TIMER_LIST: LazyInit<TimerList<TaskWakeupEvent>> = LazyInit::new(),
19}

21struct TaskWakeupEvent {
22    ticket_id: u64,
23    task: AxTaskRef,
24}
25
26impl TimerEvent for TaskWakeupEvent {
27    fn callback(self, _now: TimeValue) {
28        // Ignore the timer event if timeout was set but not triggered
29        // (wake up by `WaitQueue::notify()`).
30        // Judge if this timer event is still valid by checking the ticket ID.
31        if self.task.timer_ticket() != self.ticket_id {
32            // Timer ticket ID is not matched.
33            // Just ignore this timer event and return.
34            return;
35        }
36
37        // Timer ticket match.
38        let _ = select_run_queue::<NoOp>(&self.task).unblock_task(self.task, true);
39    }
40}

42fn program_next_precise_deadline(timer_list: &TimerList<TaskWakeupEvent>) {
43    let Some(deadline) = timer_list.next_deadline() else {
44        return;
45    };
46    let now = wall_time();
47    // The runtime keeps a 100Hz periodic scheduler tick.  Timed waits with a
48    // deadline before the next periodic tick need a one-shot interrupt at the
49    // actual timer-list deadline; otherwise short POSIX waits are rounded up to
50    // the next 10ms tick.  Do not reprogram far-future timers here: those would
51    // unnecessarily postpone the periodic scheduler tick.
52    let periodic_interval =
53        core::time::Duration::from_nanos((NANOS_PER_SEC / axconfig::TICKS_PER_SEC as u64).max(1));
54    if deadline > now + periodic_interval {
55        return;
56    }
57
58    let monotonic_deadline = deadline
59        .as_nanos()
60        .saturating_sub(epochoffset_nanos() as u128)
61        .min(u64::MAX as u128) as u64;
62    let monotonic_now = monotonic_time_nanos();
63    // Avoid programming an already-expired (or sub-millisecond race-window)
64    // one-shot deadline. Some timer backends convert absolute nanoseconds into
65    // a relative counter delta, so a deadline that expires during programming
66    // can underflow into a huge sleep and strand timed waiters.
67    let safe_deadline = if monotonic_deadline <= monotonic_now
68        || monotonic_deadline - monotonic_now < MIN_ONESHOT_DELAY_NANOS
69    {
70        monotonic_now.saturating_add(MIN_ONESHOT_DELAY_NANOS)
71    } else {
72        monotonic_deadline
73    };
74    set_oneshot_timer(safe_deadline);
75}

77pub fn set_alarm_wakeup(deadline: TimeValue, task: AxTaskRef) {
78    TIMER_LIST.with_current(|timer_list| {
79        let ticket_id = TIMER_TICKET_ID.fetch_add(1, Ordering::AcqRel);
80        task.set_timer_ticket(ticket_id);
81        timer_list.set(deadline, TaskWakeupEvent { ticket_id, task });
82        program_next_precise_deadline(timer_list);
83    })
84}
85
86pub fn cancel_alarm_wakeup(task: &AxTaskRef) {
87    TIMER_LIST.with_current(|timer_list| {
88        timer_list.cancel(|event| Arc::ptr_eq(&event.task, task));
89        program_next_precise_deadline(timer_list);
90    })
91}

93pub fn check_events() {
94    loop {
95        let now = wall_time();
96        let event = unsafe {
97            // Safety: IRQs are disabled at this time.
98            TIMER_LIST.current_ref_mut_raw()
99        }
100        .expire_one(now);
101        if let Some((_deadline, event)) = event {
102            event.callback(now);
103        } else {
104            break;
105        }
106    }
107    TIMER_LIST.with_current(|timer_list| program_next_precise_deadline(timer_list));
108}

110pub fn init() {

112        timer_list.init_once(TimerList::new());
113    });
