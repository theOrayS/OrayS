12const FD_SETSIZE: usize = 1024;
13const BITS_PER_WORD: usize = c_ulong::BITS as usize;
14const FD_SETSIZE_WORDS: usize = FD_SETSIZE.div_ceil(BITS_PER_WORD);
15const READ_SET: usize = 0;
16const WRITE_SET: usize = 1;
17const EXCEPT_SET: usize = 2;
18const FD_SET_GROUPS: usize = 3;
19
20struct FdSets {
21    nfds: usize,
22    bits: [[c_ulong; FD_SETSIZE_WORDS]; FD_SET_GROUPS],
23}
24
25fn timeval_to_duration(tv: ctypes::timeval) -> LinuxResult<core::time::Duration> {
26    if tv.tv_sec < 0 || tv.tv_usec < 0 || tv.tv_usec > 999_999 {
27        return Err(LinuxError::EINVAL);
28    }
29    Ok(tv.into())
30}
31
32impl FdSets {
33    fn empty(nfds: usize) -> Self {
34        Self {
35            nfds,
36            bits: [[0; FD_SETSIZE_WORDS]; FD_SET_GROUPS],
37        }
38    }
39
40    fn from(
41        nfds: usize,
42        read_fds: *const ctypes::fd_set,
43        write_fds: *const ctypes::fd_set,
44        except_fds: *const ctypes::fd_set,
45    ) -> LinuxResult<Self> {
46        let mut sets = Self::empty(nfds);
47        sets.copy_from_fd_set(READ_SET, read_fds)?;
48        sets.copy_from_fd_set(WRITE_SET, write_fds)?;
49        sets.copy_from_fd_set(EXCEPT_SET, except_fds)?;
50        Ok(sets)
51    }
52
53    fn nfds_words(&self) -> usize {
54        self.nfds.div_ceil(BITS_PER_WORD)
55    }
56
57    fn clear(&mut self) {
58        let words = self.nfds_words();
59        for set in &mut self.bits {
60            set[..words].fill(0);
61        }
62    }
63
64    fn copy_from_fd_set(&mut self, set_idx: usize, fds: *const ctypes::fd_set) -> LinuxResult {
65        if fds.is_null() {
66            return Ok(());
67        }
68        let words = self.nfds_words();
69        let src = unsafe { read_user_value(fds)? };
70        self.bits[set_idx][..words].copy_from_slice(&src.fds_bits[..words]);
71        Ok(())
72    }
73
74    fn set_fd(&mut self, set_idx: usize, fd: usize) {
75        self.bits[set_idx][fd / BITS_PER_WORD] |= 1 << (fd % BITS_PER_WORD);
76    }
77
78    unsafe fn write_back_to(
79        &self,
80        read_fds: *mut ctypes::fd_set,
81        write_fds: *mut ctypes::fd_set,
82        except_fds: *mut ctypes::fd_set,
83    ) -> LinuxResult {
84        unsafe { self.copy_to_fd_set(READ_SET, read_fds)? };
85        unsafe { self.copy_to_fd_set(WRITE_SET, write_fds)? };
86        unsafe { self.copy_to_fd_set(EXCEPT_SET, except_fds)? };
87        Ok(())
88    }
89
90    unsafe fn copy_to_fd_set(&self, set_idx: usize, fds: *mut ctypes::fd_set) -> LinuxResult {
91        if fds.is_null() {
92            return Ok(());
93        }
94        let words = self.nfds_words();
95        let mut dst = unsafe { read_user_value(fds as *const ctypes::fd_set)? };
96        dst.fds_bits[..words].copy_from_slice(&self.bits[set_idx][..words]);
97        unsafe { write_user_value(fds, dst)? };
98        Ok(())
99    }
100
101    fn poll_all(&self, result_sets: &mut FdSets) -> LinuxResult<usize> {
102        result_sets.clear();
103        let mut res_num = 0;
104        for word_idx in 0..self.nfds_words() {
105            let read_bits = self.bits[READ_SET][word_idx];
106            let write_bits = self.bits[WRITE_SET][word_idx];
107            let except_bits = self.bits[EXCEPT_SET][word_idx];
108
109            let all_bits = read_bits | write_bits | except_bits;
110            if all_bits == 0 {
111                continue;
112            }
113            let fd_base = word_idx * BITS_PER_WORD;
114            let valid_bits = self.nfds.saturating_sub(fd_base).min(BITS_PER_WORD);
115            let mut bits = if valid_bits < BITS_PER_WORD {
116                all_bits & ((1 << valid_bits) - 1)
117            } else {
118                all_bits
119            };
120            while bits != 0 {
121                let bit_idx = bits.trailing_zeros() as usize;
122                let bit = 1 << bit_idx;
123                let fd = fd_base + bit_idx;
124                match get_file_like(fd as _)?.poll() {
125                    Ok(state) => {
126                        if state.readable && read_bits & bit != 0 {
127                            result_sets.set_fd(READ_SET, fd);
128                            res_num += 1;
129                        }
130                        if state.writable && write_bits & bit != 0 {
131                            result_sets.set_fd(WRITE_SET, fd);
132                            res_num += 1;
133                        }
134                    }
135                    Err(e) => {
136                        debug!("    except: {} {:?}", fd, e);
137                        if except_bits & bit != 0 {
138                            result_sets.set_fd(EXCEPT_SET, fd);
139                            res_num += 1;
140                        }
141                    }
142                }
143                bits &= bits - 1;
144            }
145        }
146        Ok(res_num)
147    }
148}

150/// Monitor multiple file descriptors, waiting until one or more of the file descriptors become "ready" for some class of I/O operation
151///
152/// # Safety
153///
154/// Any non-null fd-set pointer must be valid for both reads and writes of the
155/// fd-set words covered by `nfds`; a non-null `timeout` must be valid for reads
156/// of one `timeval`.
157pub unsafe fn sys_select(
158    nfds: c_int,
159    readfds: *mut ctypes::fd_set,
160    writefds: *mut ctypes::fd_set,
161    exceptfds: *mut ctypes::fd_set,
162    timeout: *mut ctypes::timeval,
163) -> c_int {
164    debug!(
165        "sys_select <= {} {:#x} {:#x} {:#x}",
166        nfds, readfds as usize, writefds as usize, exceptfds as usize
167    );
168    syscall_body!(sys_select, {
169        if nfds < 0 || nfds as usize > FD_SETSIZE {
170            return Err(LinuxError::EINVAL);
171        }
172        let nfds = nfds as usize;
173        let deadline = if timeout.is_null() {
174            None
175        } else {
176            let tv = unsafe { read_user_value(timeout as *const ctypes::timeval)? };
177            Some(wall_time() + timeval_to_duration(tv)?)
178        };
179        let fd_sets = FdSets::from(nfds, readfds, writefds, exceptfds)?;
180        let mut result_sets = FdSets::empty(nfds);
181
182        loop {
183            #[cfg(feature = "net")]
184            axnet::poll_interfaces();
185            let res = match fd_sets.poll_all(&mut result_sets) {
186                Ok(res) => res,
187                Err(err) => {
188                    unsafe { result_sets.write_back_to(readfds, writefds, exceptfds)? };
189                    return Err(err);
190                }
191            };
192            if res > 0 {
193                unsafe { result_sets.write_back_to(readfds, writefds, exceptfds)? };
194                return Ok(res);
195            }
196
197            if super::wait_for_poll_retry(deadline) {
198                debug!("    timeout!");
199                unsafe { result_sets.write_back_to(readfds, writefds, exceptfds)? };
200                return Ok(0);
201            }
202        }
203    })
