8fn current_rlimit(resource: u32) -> Option<ctypes::rlimit> {
9    match resource {
10        ctypes::RLIMIT_STACK => Some(ctypes::rlimit {
11            rlim_cur: axconfig::TASK_STACK_SIZE as _,
12            rlim_max: axconfig::TASK_STACK_SIZE as _,
13        }),
14        #[cfg(feature = "fd")]
15        ctypes::RLIMIT_NOFILE => Some(ctypes::rlimit {
16            rlim_cur: super::fd_ops::AX_FILE_LIMIT as _,
17            rlim_max: super::fd_ops::AX_FILE_LIMIT as _,
18        }),
19        _ => None,
20    }
21}

23unsafe fn write_rlimit_output(rlimits: *mut ctypes::rlimit, value: ctypes::rlimit) -> LinuxResult {
24    unsafe { write_user_value(rlimits, value) }

27/// Get resource limitations
28///
29/// TODO: support more resource types
30///
31/// # Safety
32///
33/// `rlimits` must be writable for one `rlimit` value when non-null.
34pub unsafe fn sys_getrlimit(resource: c_int, rlimits: *mut ctypes::rlimit) -> c_int {
35    debug!("sys_getrlimit <= {} {:#x}", resource, rlimits as usize);
36    syscall_body!(sys_getrlimit, {
37        match resource as u32 {
38            ctypes::RLIMIT_STACK => {}
39            ctypes::RLIMIT_NOFILE => {}
40            _ => return Err::<c_int, LinuxError>(LinuxError::EINVAL),
41        }
42        if rlimits.is_null() {
43            return Err(LinuxError::EFAULT);
44        }
45        if let Some(limit) = current_rlimit(resource as u32) {
46            unsafe { write_rlimit_output(rlimits, limit)? };
47            Ok(0)
48        } else {
49            Err(LinuxError::ENOSYS)
50        }
51    })
52}

63    syscall_body!(sys_setrlimit, {
64        match resource as u32 {
65            crate::ctypes::RLIMIT_STACK => {}
66            crate::ctypes::RLIMIT_NOFILE => {}
67            _ => return Err::<c_int, LinuxError>(LinuxError::EINVAL),
68        }
69        if rlimits.is_null() {
70            return Err(LinuxError::EFAULT);
71        }
72        check_null_ptr(rlimits)?;
73        // Currently do not support changing resource limits.
74        Err::<c_int, LinuxError>(LinuxError::ENOSYS)
