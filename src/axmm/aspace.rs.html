97    ///
98    /// It copies the page table entries only rather than the memory regions,

104        if self.va_range.overlaps(other.va_range) {
105            return ax_err!(InvalidInput, "address space overlap");
106        }
107        self.pt
108            .cursor()
109            .copy_from(&other.pt, other.base(), other.size());

113    /// Clones all user mappings from another address space.
114    ///
115    /// Unlike [`Self::copy_mappings_from`], this duplicates user mappings even

117    /// backends are recreated in the destination. Resident pages are retained

135    fn clone_user_mappings_from_inner(
136        &mut self,
137        other: &mut AddrSpace,
138        cow_writable: bool,
139    ) -> AxResult {
140        if self.va_range != other.va_range {
141            return ax_err!(InvalidInput, "address space range mismatch");
142        }
143
144        self.clear();
145
146        let mut parent_tlb_flush_needed = false;
147        for area in other.areas.iter() {
148            let mut shared_pages = BTreeMap::new();
149            let mut retained_frames = Vec::new();
150            let mut parent_protect_pages = Vec::new();
151            let alloc_missing = area.backend().alloc_missing_on_fault();
152            let cow_pages =
153                cow_writable && area.flags().contains(MappingFlags::WRITE) && alloc_missing;
154            for vaddr in PageIter4K::new(area.start(), area.end())
155                .expect("memory area bounds must be 4K aligned")
156            {
157                let parent_pte = other.pt.query(vaddr).ok();
158                let query = parent_pte.or_else(|| {
159                    area.backend()
160                        .shared_page(vaddr)
161                        .map(|(paddr, flags)| (paddr, flags, PageSize::Size4K))
162                });
163                let Some((src_paddr, src_flags, page_size)) = query else {
164                    continue;
165                };
166                if page_size.is_huge() {
167                    warn!(
168                        "clone_user_mappings_from: huge user page is not supported area_start={:#x} area_end={:#x} vaddr={:#x} src_paddr={:#x} flags={:?} backend={}",
169                        area.start(),
170                        area.end(),
171                        vaddr,
172                        src_paddr,
173                        src_flags,
174                        area.backend().kind_name()
175                    );
176                    self.clear();
177                    return ax_err!(BadState, "failed to clone huge user page");
178                }
179                let mut child_flags = src_flags;
180                if cow_pages {
181                    child_flags.remove(MappingFlags::WRITE);
182                    if parent_pte.is_some() {
183                        parent_protect_pages.push((vaddr, child_flags));
184                    }
185                }
186                shared_pages.insert(vaddr.as_usize(), (src_paddr, child_flags));
187                retained_frames.push(src_paddr);
188            }
189
190            let cloned_area = MemoryArea::new(
191                area.start(),
192                area.size(),
193                area.flags(),
194                Backend::new_shared(shared_pages.clone(), alloc_missing),
195            );
196            if let Err(err) = self.areas.map(cloned_area, &mut self.pt, false) {
197                warn!(
198                    "clone_user_mappings_from: map cloned area failed start={:#x} end={:#x} size={:#x} flags={:?} backend={} err={:?}",
199                    area.start(),
200                    area.end(),
201                    area.size(),
202                    area.flags(),
203                    area.backend().kind_name(),
204                    err
205                );
206                self.clear();
207                return Err(mapping_err_to_ax_err(err));
208            }
209
210            for frame in retained_frames {
211                retain_shared_frame(frame);
212            }
213
214            for (vaddr, flags) in parent_protect_pages {
215                if let Err(err) = other.pt.cursor().protect(vaddr, flags) {
216                    if parent_tlb_flush_needed {
217                        axhal::asm::flush_tlb(None);
218                    }
219                    warn!(
220                        "clone_user_mappings_from: COW parent protect failed area_start={:#x} area_end={:#x} vaddr={:#x} flags={:?} err={:?}",
221                        area.start(),
222                        area.end(),
223                        vaddr,
224                        flags,
225                        err
226                    );
227                    self.clear();
228                    return ax_err!(BadState, "failed to protect parent COW page");
229                }
230                parent_tlb_flush_needed = true;
231            }
232        }
233        if parent_tlb_flush_needed {
234            axhal::asm::flush_tlb(None);
235        }
236        Ok(())
237    }

253    /// Add a new linear mapping.
254    ///
255    /// See [`Backend`] for more details about the mapping backends.
256    ///
257    /// The `flags` parameter indicates the mapping permissions and attributes.
258    ///
259    /// Returns an error if the address range is out of the address space or not
260    /// aligned.
261    pub fn map_linear(
262        &mut self,
263        start_vaddr: VirtAddr,
264        start_paddr: PhysAddr,
265        size: usize,
266        flags: MappingFlags,
267    ) -> AxResult {
268        if !self.contains_range(start_vaddr, size) {
269            return ax_err!(InvalidInput, "address out of range");
270        }
271        if !start_vaddr.is_aligned_4k() || !start_paddr.is_aligned_4k() || !is_aligned_4k(size) {
272            return ax_err!(InvalidInput, "address not aligned");
273        }
274
275        let offset = start_vaddr.as_usize() - start_paddr.as_usize();
276        let area = MemoryArea::new(start_vaddr, size, flags, Backend::new_linear(offset));
277        self.areas
278            .map(area, &mut self.pt, false)
279            .map_err(mapping_err_to_ax_err)?;
280        Ok(())
281    }

356    /// To process data in this area with the given function.
357    ///
358    /// Now it supports reading and writing data in the given interval.
359    fn process_area_data<F>(
360        &mut self,
361        start: VirtAddr,
362        size: usize,
363        access_flags: MappingFlags,
364        fault_flags: PageFaultFlags,
365        mut f: F,
366    ) -> AxResult
367    where
368        F: FnMut(VirtAddr, usize, usize),
369    {
370        if !self.contains_range(start, size) {
371            return ax_err!(InvalidInput, "address out of range");
372        }
373        let mut cnt = 0;
374        // If start is aligned to 4K, start_align_down will be equal to start_align_up.
375        let end_align_up = (start + size).align_up_4k();
376        for vaddr in PageIter4K::new(start.align_down_4k(), end_align_up)
377            .expect("Failed to create page iterator")
378        {
379            let (mut paddr, _, _) = self.ensure_page_access(vaddr, access_flags, fault_flags)?;
380
381            let mut copy_size = (size - cnt).min(PAGE_SIZE_4K);
382
383            if copy_size == 0 {
384                break;
385            }
386            if vaddr == start.align_down_4k() && start.align_offset_4k() != 0 {
387                let align_offset = start.align_offset_4k();
388                copy_size = copy_size.min(PAGE_SIZE_4K - align_offset);
389                paddr += align_offset;
390            }
391            f(phys_to_virt(paddr), cnt, copy_size);
392            cnt += copy_size;
393        }
394        Ok(())
395    }

397    fn ensure_page_access(
398        &mut self,
399        vaddr: VirtAddr,
400        access_flags: MappingFlags,
401        fault_flags: PageFaultFlags,
402    ) -> AxResult<(PhysAddr, MappingFlags, PageSize)> {
403        let needs_write = access_flags.contains(MappingFlags::WRITE);
404        match self.pt.query(vaddr) {
405            Ok((paddr, flags, page_size)) => {
406                if needs_write && !flags.contains(MappingFlags::WRITE) {
407                    if self.handle_page_fault(vaddr, PageFaultFlags::WRITE) {
408                        return self.ensure_write_resolved(vaddr);
409                    }
410                    if !self.area_allows_write(vaddr) {
411                        return Ok((paddr, flags, page_size));
412                    }
413                    return Err(AxError::BadAddress);
414                }
415                if !needs_write && !flags.contains(access_flags) {
416                    return Err(AxError::BadAddress);
417                }
418                Ok((paddr, flags, page_size))
419            }
420            _ => {
421                let handled = self.handle_page_fault(vaddr, fault_flags)
422                    || (needs_write && self.handle_page_fault(vaddr, PageFaultFlags::READ));
423                if !handled {
424                    return Err(AxError::BadAddress);
425                }
426                if needs_write {
427                    return self.ensure_write_resolved(vaddr);
428                }
429                let (paddr, flags, page_size) =
430                    self.pt.query(vaddr).map_err(|_| AxError::BadAddress)?;
431                if !flags.contains(access_flags) {
432                    return Err(AxError::BadAddress);
433                }
434                Ok((paddr, flags, page_size))
435            }
436        }
437    }

439    fn ensure_write_resolved(
440        &mut self,
441        vaddr: VirtAddr,
442    ) -> AxResult<(PhysAddr, MappingFlags, PageSize)> {
443        let (paddr, flags, page_size) = self.pt.query(vaddr).map_err(|_| AxError::BadAddress)?;
444        if flags.contains(MappingFlags::WRITE) || !self.area_allows_write(vaddr) {
445            return Ok((paddr, flags, page_size));
446        }
447        if self.handle_page_fault(vaddr, PageFaultFlags::WRITE) {
448            let (paddr, flags, page_size) =
449                self.pt.query(vaddr).map_err(|_| AxError::BadAddress)?;
450            if flags.contains(MappingFlags::WRITE) {
451                return Ok((paddr, flags, page_size));
452            }
453        }
454        ax_err!(BadAddress, "write fault was not resolved")
455    }

463    /// To read data from the address space.
464    ///
465    /// # Arguments
466    ///
467    /// * `start` - The start virtual address to read.
468    /// * `buf` - The buffer to store the data.
469    pub fn read(&mut self, start: VirtAddr, buf: &mut [u8]) -> AxResult {
470        self.process_area_data(
471            start,
472            buf.len(),
473            MappingFlags::READ,
474            PageFaultFlags::READ,
475            |src, offset, read_size| unsafe {
476                core::ptr::copy_nonoverlapping(
477                    src.as_ptr(),
478                    buf.as_mut_ptr().add(offset),
479                    read_size,
480                );
481            },
482        )
483    }

507    /// Populates lazy pages in the specified virtual address range.
508    pub fn populate_range(
509        &mut self,
510        start: VirtAddr,
511        size: usize,
512        access_flags: PageFaultFlags,
513    ) -> AxResult {
514        if !self.contains_range(start, size) {
515            return ax_err!(InvalidInput, "address out of range");
516        }
517        let end_align_up = (start + size).align_up_4k();
518        for vaddr in PageIter4K::new(start.align_down_4k(), end_align_up)
519            .expect("Failed to create page iterator")
520        {
521            if self.pt.query(vaddr).is_err() && !self.handle_page_fault(vaddr, access_flags) {
522                return ax_err!(BadState, "failed to populate lazy page");
523            }
524        }
525        Ok(())
526    }

551    /// Checks whether an access to the specified memory region is valid.
552    ///
553    /// Returns `true` if the memory region given by `range` is all mapped and
554    /// has proper permission flags (i.e. containing `access_flags`).
555    pub fn can_access_range(
556        &self,
557        start: VirtAddr,
558        size: usize,
559        access_flags: MappingFlags,
560    ) -> bool {
561        let mut range = VirtAddrRange::from_start_size(start, size);
562        for area in self.areas.iter() {
563            if area.end() <= range.start {
564                continue;
565            }
566            if area.start() > range.start {
567                return false;
568            }
569
570            // This area overlaps with the memory region
571            if !area.flags().contains(access_flags) {
572                return false;
573            }
574
575            range.start = area.end();
576            if range.is_empty() {
577                return true;
578            }
579        }
580
581        false
582    }

584    /// Returns diagnostic information for one virtual address.
585    pub fn query_address(&self, vaddr: VirtAddr) -> AddrSpaceQuery {
586        let contains = self.va_range.contains(vaddr);
587        let mut query = AddrSpaceQuery {
588            contains,
589            area_found: false,
590            area_start: 0,
591            area_end: 0,
592            area_flags: MappingFlags::empty(),
593            backend: "none",
594            pte_mapped: false,
595            paddr: 0,
596            pte_flags: MappingFlags::empty(),
597            page_size: None,
598            shared_metadata: false,
599        };
600        if let Some(area) = self.areas.find(vaddr) {
601            query.area_found = true;
602            query.area_start = area.start().as_usize();
603            query.area_end = area.end().as_usize();
604            query.area_flags = area.flags();
605            query.backend = area.backend().kind_name();
606            query.shared_metadata = area.backend().shared_page(vaddr.align_down_4k()).is_some();
607        }
608        if let Ok((paddr, flags, page_size)) = self.pt.query(vaddr.align_down_4k()) {
609            query.pte_mapped = true;
610            query.paddr = paddr.as_usize();
611            query.pte_flags = flags;
612            query.page_size = Some(page_size);
613        }
614        query
615    }

617    /// Handles a page fault at the given address.
618    ///
619    /// `access_flags` indicates the access type that caused the page fault.
620    ///
621    /// Returns `true` if the page fault is handled successfully (not a real
622    /// fault).
623    pub fn handle_page_fault(&mut self, vaddr: VirtAddr, access_flags: PageFaultFlags) -> bool {
624        if !self.va_range.contains(vaddr) {
625            return false;
626        }
627        if let Some(area) = self.areas.find(vaddr) {
628            let orig_flags = area.flags();
629            let access_flags = MappingFlags::from_bits_truncate(access_flags.bits());
630            if orig_flags.contains(access_flags) {
631                return area
632                    .backend()
633                    .handle_page_fault(vaddr, orig_flags, &mut self.pt);
634            }
635        }
636        false
637    }
