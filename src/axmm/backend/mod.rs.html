18#[inline]
19fn pte_flags_for_mapping(flags: MappingFlags) -> MappingFlags {
20    #[cfg(target_arch = "loongarch64")]
21    {
22        const ACCESS: MappingFlags = MappingFlags::READ
23            .union(MappingFlags::WRITE)
24            .union(MappingFlags::EXECUTE);
25        if flags.contains(MappingFlags::USER) && !flags.intersects(ACCESS) {
26            // LoongArch QEMU does not reliably raise a non-readable exception
27            // for user pages encoded only as PLV3 + NR/NX. Keep the leaf
28            // present so lifecycle paths can free/copy the frame, but remove
29            // user accessibility so a PLV3 access traps as SIGSEGV. The owning
30            // memory area still carries the original PROT_NONE flags.
31            return MappingFlags::READ;
32        }
33    }
34    flags
35}

37/// A unified enum type for different memory mapping backends.
38///
39/// Currently, three backends are implemented:
40///
41/// - **Linear**: used for linear mappings. The target physical frames are
42///   contiguous and their addresses should be known when creating the mapping.
43/// - **Allocation**: used in general, or for lazy mappings. The target physical
44///   frames are obtained from the global allocator.
45#[derive(Clone)]
46pub enum Backend {
47    /// Linear mapping backend.
48    ///
49    /// The offset between the virtual address and the physical address is
50    /// constant, which is specified by `pa_va_offset`. For example, the virtual
51    /// address `vaddr` is mapped to the physical address `vaddr - pa_va_offset`.
52    Linear {
53        /// `vaddr - paddr`.
54        pa_va_offset: usize,
55    },
56    /// Allocation mapping backend.
57    ///
58    /// If `populate` is `true`, all physical frames are allocated when the
59    /// mapping is created, and no page faults are triggered during the memory
60    /// access. Otherwise, the physical frames are allocated on demand (by
61    /// handling page faults).
62    Alloc {
63        /// Whether to populate the physical frames when creating the mapping.
64        populate: bool,
65    },
66    /// Shared physical mappings owned by another address space.
67    Shared {
68        /// Shared frame table of the owning address space.
69        pages: SharedPages,
70        /// Whether frames absent from the table may be allocated privately.
71        alloc_missing: bool,
72    },
73}
74
75impl MappingBackend for Backend {
76    type Addr = VirtAddr;
77    type Flags = MappingFlags;
78    type PageTable = PageTable;
79    fn map(&self, start: VirtAddr, size: usize, flags: MappingFlags, pt: &mut PageTable) -> bool {
80        match *self {
81            Self::Linear { pa_va_offset } => self.map_linear(start, size, flags, pt, pa_va_offset),
82            Self::Alloc { populate } => self.map_alloc(start, size, flags, pt, populate),
83            Self::Shared { ref pages, .. } => self.map_shared(start, size, pt, pages),
84        }
85    }
86
87    fn unmap(&self, start: VirtAddr, size: usize, pt: &mut PageTable) -> bool {
88        match *self {
89            Self::Linear { pa_va_offset } => self.unmap_linear(start, size, pt, pa_va_offset),
90            Self::Alloc { populate } => self.unmap_alloc(start, size, pt, populate),
91            Self::Shared { ref pages, .. } => self.unmap_shared(start, size, pt, pages),
92        }
93    }
94
95    fn protect(
96        &self,
97        start: Self::Addr,
98        size: usize,
99        new_flags: Self::Flags,
100        page_table: &mut Self::PageTable,
101    ) -> bool {
102        let pte_flags = pte_flags_for_mapping(new_flags);
103        page_table
104            .cursor()
105            .protect_region(start, size, pte_flags)
106            .is_ok()
107    }
108}

110impl Backend {
111    pub(crate) fn kind_name(&self) -> &'static str {
112        match *self {
113            Self::Linear { .. } => "linear",
114            Self::Alloc { populate: true } => "alloc-populate",
115            Self::Alloc { populate: false } => "alloc-lazy",
116            Self::Shared { .. } => "shared",
117        }
118    }
119
120    pub(crate) fn alloc_missing_on_fault(&self) -> bool {
121        match *self {
122            Self::Alloc { .. } => true,
123            Self::Shared { alloc_missing, .. } => alloc_missing,
124            Self::Linear { .. } => false,
125        }
126    }
127
128    pub(crate) fn shared_page(&self, vaddr: VirtAddr) -> Option<(PhysAddr, MappingFlags)> {
129        match *self {
130            Self::Shared { ref pages, .. } => pages.lock().get(&vaddr.as_usize()).copied(),
131            _ => None,
132        }
133    }
134
135    pub(crate) fn handle_page_fault(
136        &self,
137        vaddr: VirtAddr,
138        orig_flags: MappingFlags,
139        page_table: &mut PageTable,
140    ) -> bool {
141        match *self {
142            Self::Linear { .. } => false, // Linear mappings should not trigger page faults.
143            Self::Alloc { populate } => {
144                self.handle_page_fault_alloc(vaddr, orig_flags, page_table, populate)
145            }
146            Self::Shared {
147                ref pages,
148                alloc_missing,
149            } => self.handle_page_fault_shared(vaddr, orig_flags, page_table, pages, alloc_missing),
150        }
151    }
152}
