14static SHARED_FRAMES: LazyInit<SpinNoIrq<BTreeMap<usize, usize>>> = LazyInit::new();

16/// Reference-count statistics for shared physical frames.
17#[derive(Clone, Copy, Debug, Eq, PartialEq)]
18pub struct SharedFrameStats {
19    /// Number of tracked shared frames.
20    pub entries: usize,
21    /// Sum of reference counts over all tracked frames.
22    pub total_refs: usize,
23    /// Highest reference count of any tracked frame.
24    pub max_refcount: usize,
25}
26
27fn shared_frames() -> &'static SpinNoIrq<BTreeMap<usize, usize>> {
28    let _ = SHARED_FRAMES.call_once(|| SpinNoIrq::new(BTreeMap::new()));
29    &SHARED_FRAMES
30}
31
32pub(crate) fn shared_frame_stats() -> SharedFrameStats {
33    let frames = shared_frames().lock();
34    let mut total_refs = 0usize;
35    let mut max_refcount = 0usize;
36    for count in frames.values().copied() {
37        total_refs = total_refs.saturating_add(count);
38        max_refcount = max_refcount.max(count);
39    }
40    SharedFrameStats {
41        entries: frames.len(),
42        total_refs,
43        max_refcount,
44    }
45}

47fn alloc_frame(zeroed: bool) -> Option<PhysAddr> {
48    let vaddr = VirtAddr::from(global_allocator().alloc_pages(1, PAGE_SIZE_4K).ok()?);
49    if zeroed {
50        unsafe { core::ptr::write_bytes(vaddr.as_mut_ptr(), 0, PAGE_SIZE_4K) };
51    }
52    let paddr = virt_to_phys(vaddr);
53    Some(paddr)
54}
55
56fn dealloc_frame(frame: PhysAddr) {
57    let vaddr = phys_to_virt(frame);
58    global_allocator().dealloc_pages(vaddr.as_usize(), 1);
59}

61pub(crate) fn retain_shared_frame(frame: PhysAddr) {
62    let mut frames = shared_frames().lock();
63    let key = frame.as_usize();
64    if let Some(count) = frames.get_mut(&key) {
65        *count += 1;
66    } else {
67        frames.insert(key, 2);
68    }
69}
70
71fn shared_frame_count(frame: PhysAddr) -> Option<usize> {
72    shared_frames().lock().get(&frame.as_usize()).copied()
73}
74
75fn forget_shared_frame(frame: PhysAddr) {
76    shared_frames().lock().remove(&frame.as_usize());
77}
78
79pub(crate) fn release_owned_frame(frame: PhysAddr) {
80    let should_dealloc = {
81        let mut frames = shared_frames().lock();
82        let key = frame.as_usize();
83        match frames.get_mut(&key) {
84            Some(count) if *count > 1 => {
85                *count -= 1;
86                false
87            }
88            Some(_) => {
89                frames.remove(&key);
90                true
91            }
92            None => true,
93        }
94    };
95    if should_dealloc {
96        dealloc_frame(frame);
97    }
98}

116    pub(crate) fn map_alloc(
117        &self,
118        start: VirtAddr,
119        size: usize,
120        flags: MappingFlags,
121        pt: &mut PageTable,
122        populate: bool,
123    ) -> bool {
124        debug!(
125            "map_alloc: [{:#x}, {:#x}) {:?} (populate={})",
126            start,
127            start + size,
128            flags,
129            populate
130        );
131        if populate {
132            // allocate all possible physical frames for populated mapping.
133            let mut cursor = pt.cursor();
134            for addr in PageIter4K::new(start, start + size).unwrap() {
135                let Some(frame) = alloc_frame(true) else {
136                    let stats = frame_allocator_stats();
137                    warn!(
138                        "map_alloc: frame allocation failed start={:#x} size={:#x} at={:#x} flags={:?} populate=true free_frames={} allocated_frames={}",
139                        start, size, addr, flags, stats.free_frames, stats.allocated_frames
140                    );
141                    for rollback_addr in PageIter4K::new(start, addr).unwrap() {
142                        if let Ok((mapped_frame, _, page_size)) = cursor.unmap(rollback_addr)
143                            && !page_size.is_huge()
144                        {
145                            release_owned_frame(mapped_frame);
146                        }
147                    }
148                    return false;
149                };
150                let pte_flags = pte_flags_for_mapping(flags);
151                if cursor
152                    .map(addr, frame, PageSize::Size4K, pte_flags)
153                    .is_err()
154                {
155                    warn!(
156                        "map_alloc: page-table map failed start={:#x} size={:#x} at={:#x} frame={:#x} flags={:?} populate=true",
157                        start, size, addr, frame, flags
158                    );
159                    for rollback_addr in PageIter4K::new(start, addr).unwrap() {
160                        if let Ok((mapped_frame, _, page_size)) = cursor.unmap(rollback_addr)
161                            && !page_size.is_huge()
162                        {
163                            release_owned_frame(mapped_frame);
164                        }
165                    }
166                    release_owned_frame(frame);
167                    return false;
168                }
169            }
170            true
171        } else {
172            // Leave page-table entries absent until the first page fault.
173            true
174        }
175    }

177    pub(crate) fn unmap_alloc(
178        &self,
179        start: VirtAddr,
180        size: usize,
181        pt: &mut PageTable,
182        _populate: bool,
183    ) -> bool {
184        debug!("unmap_alloc: [{:#x}, {:#x})", start, start + size);
185        for addr in PageIter4K::new(start, start + size).unwrap() {
186            if let Ok((frame, _, page_size)) = pt.cursor().unmap(addr) {
187                // Deallocate the physical frame if there is a mapping in the
188                // page table.
189                if page_size.is_huge() {
190                    return false;
191                }
192                release_owned_frame(frame);
193            }
194        }
195        true
196    }

198    pub(crate) fn unmap_shared(
199        &self,
200        start: VirtAddr,
201        size: usize,
202        pt: &mut PageTable,
203        pages: &SharedPages,
204    ) -> bool {
205        for addr in PageIter4K::new(start, start + size).unwrap() {
206            if let Ok((frame, _, page_size)) = pt.cursor().unmap(addr) {
207                if page_size.is_huge() {
208                    return false;
209                }
210                let shared_frame = pages
211                    .lock()
212                    .get(&addr.as_usize())
213                    .is_some_and(|(source_frame, _)| *source_frame == frame);
214                if !shared_frame {
215                    release_owned_frame(frame);
216                }
217            }
218        }
219        let mut pages = pages.lock();
220        let keys = pages
221            .range(start.as_usize()..(start + size).as_usize())
222            .map(|(addr, _)| *addr)
223            .collect::<Vec<_>>();
224        for key in keys {
225            if let Some((frame, _)) = pages.remove(&key) {
226                release_owned_frame(frame);
227            }
228        }
229        true
230    }

232    pub(crate) fn map_shared(
233        &self,
234        start: VirtAddr,
235        size: usize,
236        pt: &mut PageTable,
237        pages: &SharedPages,
238    ) -> bool {
239        let end = start + size;
240        let entries = pages
241            .lock()
242            .range(start.as_usize()..end.as_usize())
243            .map(|(addr, (frame, flags))| (VirtAddr::from(*addr), *frame, *flags))
244            .collect::<Vec<_>>();
245
246        let mut cursor = pt.cursor();
247        for (addr, frame, flags) in entries.iter().copied() {
248            if cursor.map(addr, frame, PageSize::Size4K, flags).is_err() {
249                warn!(
250                    "map_shared: page-table map failed start={:#x} size={:#x} at={:#x} frame={:#x} flags={:?}",
251                    start, size, addr, frame, flags
252                );
253                for (rollback_addr, _, _) in entries
254                    .iter()
255                    .copied()
256                    .take_while(|(rollback_addr, _, _)| *rollback_addr != addr)
257                {
258                    let _ = cursor.unmap(rollback_addr);
259                }
260                return false;
261            }
262        }
263        true
264    }

266    fn handle_cow_fault(
267        &self,
268        vaddr: VirtAddr,
269        orig_flags: MappingFlags,
270        pt: &mut PageTable,
271    ) -> bool {
272        let Ok((old_frame, flags, page_size)) = pt.query(vaddr) else {
273            return false;
274        };
275        if page_size.is_huge()
276            || flags.contains(MappingFlags::WRITE)
277            || !orig_flags.contains(MappingFlags::WRITE)
278        {
279            return false;
280        }
281        let count = shared_frame_count(old_frame).unwrap_or(1);
282        if count <= 1 {
283            let res = pt.cursor().protect(vaddr, orig_flags);
284            if let Err(err) = &res {
285                warn!(
286                    "handle_page_fault_alloc: COW protect failed for {:#x}: {:?}",
287                    vaddr, err
288                );
289            } else {
290                forget_shared_frame(old_frame);
291                axhal::asm::flush_tlb(Some(vaddr));
292            }
293            return res.is_ok();
294        }
295        let Some(new_frame) = alloc_frame(false) else {
296            let stats = frame_allocator_stats();
297            warn!(
298                "handle_page_fault_alloc: COW frame allocation failed for {:#x} flags={:?} free_frames={} allocated_frames={}",
299                vaddr, orig_flags, stats.free_frames, stats.allocated_frames
300            );
301            return false;
302        };
303        unsafe {
304            core::ptr::copy_nonoverlapping(
305                phys_to_virt(old_frame).as_ptr(),
306                phys_to_virt(new_frame).as_mut_ptr(),
307                PAGE_SIZE_4K,
308            );
309        }
310        let res = pt.cursor().remap(vaddr, new_frame, orig_flags);
311        if let Err(err) = &res {
312            warn!(
313                "handle_page_fault_alloc: COW remap failed for {:#x}: {:?}",
314                vaddr, err
315            );
316            dealloc_frame(new_frame);
317            return false;
318        }
319        axhal::asm::flush_tlb(Some(vaddr));
320        release_owned_frame(old_frame);
321        true
322    }

324    fn map_fresh_frame(&self, vaddr: VirtAddr, flags: MappingFlags, pt: &mut PageTable) -> bool {
325        if let Some(frame) = alloc_frame(true) {
326            let pte_flags = pte_flags_for_mapping(flags);
327            let res = pt.cursor().map(vaddr, frame, PageSize::Size4K, pte_flags);
328            if let Err(e) = &res {
329                warn!(
330                    "handle_page_fault_alloc: map failed for {:#x}: {:?}",
331                    vaddr, e
332                );
333                release_owned_frame(frame);
334            }
335            res.is_ok()
336        } else {
337            let stats = frame_allocator_stats();
338            warn!(
339                "handle_page_fault_alloc: frame allocation failed for {:#x} flags={:?} free_frames={} allocated_frames={}",
340                vaddr, flags, stats.free_frames, stats.allocated_frames
341            );
342            false
343        }
344    }

346    pub(crate) fn handle_page_fault_alloc(
347        &self,
348        vaddr: VirtAddr,
349        orig_flags: MappingFlags,
350        pt: &mut PageTable,
351        populate: bool,
352    ) -> bool {
353        let vaddr = vaddr.align_down_4k();
354        if pt.query(vaddr).is_ok() {
355            self.handle_cow_fault(vaddr, orig_flags, pt)
356        } else if populate {
357            false // Populated mappings should not trigger page faults.
358        } else {
359            self.map_fresh_frame(vaddr, orig_flags, pt)
360        }
361    }

363    pub(crate) fn handle_page_fault_shared(
364        &self,
365        vaddr: VirtAddr,
366        orig_flags: MappingFlags,
367        pt: &mut PageTable,
368        pages: &SharedPages,
369        alloc_missing: bool,
370    ) -> bool {
371        let vaddr = vaddr.align_down_4k();
372        if pt.query(vaddr).is_ok() {
373            let handled = self.handle_cow_fault(vaddr, orig_flags, pt);
374            if handled {
375                pages.lock().remove(&vaddr.as_usize());
376            }
377            return handled;
378        }
379        if let Some((frame, flags)) = pages.lock().get(&vaddr.as_usize()).copied() {
380            let res = pt.cursor().map(vaddr, frame, PageSize::Size4K, flags);
381            if let Err(err) = &res {
382                warn!(
383                    "handle_page_fault_shared: map shared page failed for {:#x}: {:?}",
384                    vaddr, err
385                );
386            }
387            return res.is_ok();
388        }
389        if alloc_missing {
390            self.map_fresh_frame(vaddr, orig_flags, pt)
391        } else {
392            false
393        }
394    }
