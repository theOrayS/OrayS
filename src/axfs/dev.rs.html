1use axdriver::prelude::*;
2
3const BLOCK_SIZE: usize = 512;
4
5/// A disk device with a cursor.
6pub struct Disk {
7    block_id: u64,
8    offset: usize,
9    dev: AxBlockDevice,
10    read_buffer: [u8; BLOCK_SIZE],
11    write_buffer: [u8; BLOCK_SIZE],
12    write_buffer_block_id: u64,
13    write_buffer_dirty: bool,
14}
15
16impl Disk {
17    /// Create a new disk.
18    pub fn new(dev: AxBlockDevice) -> Self {
19        assert_eq!(BLOCK_SIZE, dev.block_size());
20        Self {
21            block_id: 0,
22            offset: 0,
23            dev,
24            read_buffer: [0; BLOCK_SIZE],
25            write_buffer: [0; BLOCK_SIZE],
26            write_buffer_block_id: 0,
27            write_buffer_dirty: false,
28        }
29    }
30
31    /// Get the size of the disk.
32    pub fn size(&self) -> u64 {
33        self.dev.num_blocks() * BLOCK_SIZE as u64
34    }
35
36    /// Get the position of the cursor.
37    pub fn position(&self) -> u64 {
38        self.block_id * BLOCK_SIZE as u64 + self.offset as u64
39    }
40
41    /// Set the position of the cursor.
42    pub fn set_position(&mut self, pos: u64) {
43        self.block_id = pos / BLOCK_SIZE as u64;
44        self.offset = pos as usize % BLOCK_SIZE;
45    }
46
47    /// Flushes the pending partial-block write buffer, if any.
48    pub fn flush(&mut self) -> DevResult {
49        if self.write_buffer_dirty {
50            self.dev
51                .write_block(self.write_buffer_block_id, &self.write_buffer)?;
52            self.write_buffer_dirty = false;
53        }
54        Ok(())
55    }
56
57    fn advance(&mut self, count: usize) {
58        self.offset += count;
59        if self.offset >= BLOCK_SIZE {
60            self.block_id += 1;
61            self.offset -= BLOCK_SIZE;
62        }
63    }
64
65    fn prepare_write_buffer(&mut self) -> DevResult {
66        if self.write_buffer_dirty && self.write_buffer_block_id == self.block_id {
67            return Ok(());
68        }
69        self.flush()?;
70        self.dev.read_block(self.block_id, &mut self.write_buffer)?;
71        self.write_buffer_block_id = self.block_id;
72        self.write_buffer_dirty = true;
73        Ok(())
74    }
75
76    /// Read within one block, returns the number of bytes read.
77    pub fn read_one(&mut self, buf: &mut [u8]) -> DevResult<usize> {
78        let read_size = if self.offset == 0 && buf.len() >= BLOCK_SIZE {
79            // whole block
80            if self.write_buffer_dirty && self.write_buffer_block_id == self.block_id {
81                buf[..BLOCK_SIZE].copy_from_slice(&self.write_buffer);
82            } else {
83                self.dev
84                    .read_block(self.block_id, &mut buf[0..BLOCK_SIZE])?;
85            }
86            self.block_id += 1;
87            BLOCK_SIZE
88        } else {
89            // partial block
90            let start = self.offset;
91            let count = buf.len().min(BLOCK_SIZE - self.offset);
92
93            if self.write_buffer_dirty && self.write_buffer_block_id == self.block_id {
94                buf[..count].copy_from_slice(&self.write_buffer[start..start + count]);
95            } else {
96                self.dev.read_block(self.block_id, &mut self.read_buffer)?;
97                buf[..count].copy_from_slice(&self.read_buffer[start..start + count]);
98            }
99
100            self.advance(count);
101            count
102        };
103        Ok(read_size)
104    }
105
106    /// Write within one block, returns the number of bytes written.
107    pub fn write_one(&mut self, buf: &[u8]) -> DevResult<usize> {
108        let write_size = if self.offset == 0 && buf.len() >= BLOCK_SIZE {
109            // whole block
110            if self.write_buffer_dirty {
111                if self.write_buffer_block_id == self.block_id {
112                    self.write_buffer_dirty = false;
113                } else {
114                    self.flush()?;
115                }
116            }
117            self.dev.write_block(self.block_id, &buf[0..BLOCK_SIZE])?;
118            self.block_id += 1;
119            BLOCK_SIZE
120        } else {
121            // partial block
122            let start = self.offset;
123            let count = buf.len().min(BLOCK_SIZE - self.offset);
124
125            self.prepare_write_buffer()?;
126            self.write_buffer[start..start + count].copy_from_slice(&buf[..count]);
127
128            self.advance(count);
129            if self.offset == 0 {
130                self.flush()?;
131            }
132            count
133        };
134        Ok(write_size)
135    }
136}
